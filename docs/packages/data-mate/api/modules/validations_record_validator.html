reject

▸ reject(input: RecordInput, _parentContext: RecordInput, args: DMOptions): RecordInput | null

Will return true if an object DOES NOT match the xLucene expression

example const obj1 = { foo: 'hello', bar: 'stuff' }; const obj2 = { foo: 123412 }; const args = { query: 'exists:bar' };

const results1 = RecordValidator.reject(obj1, obj1, args); const results2 = RecordValidator.reject(obj2, obj2, args);

results1; // false; results2; // true;

required

▸ required(input: RecordInput, _parentContext: RecordInput, args: { fields: string[] }): RecordInput | null
